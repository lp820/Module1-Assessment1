#1. Print a list of all doctors based at a particular hospital.

OUTPUT ‘Please enter hospital_id: ’
hospital_id <- USERINPUT 
FOR row in doctors_table

	THEN

ENDFOR
OUTPUT list of results

patient_id <- USERINPUT

#4. Add a new patient to the database, including being registered with one of the doctors.

OUTPUT “Please enter new patient details in the following format: person_id, name, date_of_birth, address, role, doctor_id: ”
person_id, name, date_of_birth, address, role, doctor_id  <- USERINPUT
IF doctor_id IN doctors_table.person_id
	THEN
	ADD person_id, name, date_of_birth, address, role, doctor_id  TO patients_table
ELSE 
	OUTPUT “Error - unable to add record to database. Patient must be registered with a doctor.”
ENDIF



#5. Identify which doctor has made the most prescriptions.

GROUP prescription_table BY doctor_id
FOR EACH UNIQUE doctor_id
	COUNT number of prescription entries
	ADD doctor_id and count TO results
END FOR
ORDER results BY count DESCENDING
OUTPUT doctor_id at top row = doctor with highest prescription count
